# ---------- HTTP -> HTTPS redirect for all four hosts ----------
server {
    listen 80;
    listen [::]:80;
    server_name explorer-p2p.ergoplatform.com api-p2p.ergoplatform.com graphql-p2p.ergoplatform.com api-2ndtl.ergoplatform.com node-p2p.ergoplatform.com;
    return 301 https://$host$request_uri;
}

# ---------- Explorer UI (proxy to :3000) ----------
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name explorer-p2p.ergoplatform.com;

    ssl_certificate           /etc/ssl/cloudflare/ergoplatform-p2p.pem;
    ssl_certificate_key       /etc/ssl/cloudflare/ergoplatform-p2p.key;
    ssl_trusted_certificate   /etc/ssl/cloudflare/origin_ca_rsa_root.pem;

#    include snippets/ssl-params.conf;   # (optional) if you have this; otherwise omit

    location / {
        proxy_http_version 1.1;
        proxy_set_header Host              $http_host;          # pass original host
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # WebSocket / SSE friendliness (harmless if unused)
        proxy_set_header Upgrade           $http_upgrade;
        proxy_set_header Connection        "upgrade";

        proxy_pass http://localhost:3000/;

        proxy_ssl_session_reuse off;
        proxy_pass_header Server;
        proxy_cache_bypass $http_upgrade;
        proxy_redirect off;
    }
    # Serve the UI runtime config directly from Nginx (bypass the app's fallback)
    location = /config/app.config.js {
        default_type application/javascript;
        add_header Cache-Control "public, max-age=300";
        return 200 'window.APP_CONFIG = {
        API_BASE: "https://api-p2p.ergoplatform.com/api/v0",
        GRAPHQL_BASE: "https://graphql-p2p.ergoplatform.com"
    };';
   }

}

# ---------- API (proxy to :8080) with v0 rewrite ----------
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name api-p2p.ergoplatform.com api-2ndtl.ergoplatform.com;

    ssl_certificate           /etc/ssl/cloudflare/ergoplatform-p2p.pem;
    ssl_certificate_key       /etc/ssl/cloudflare/ergoplatform-p2p.key;
    ssl_trusted_certificate   /etc/ssl/cloudflare/origin_ca_rsa_root.pem;

#    include snippets/ssl-params.conf;   # (optional)

    location / {
        proxy_http_version 1.1;
        proxy_set_header Host              $http_host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Keep your compatibility rewrite:
        rewrite ^(?!/api/v[0-9])(/.*)$ /api/v0$1 break;

        proxy_pass http://localhost:8080/;
        proxy_ssl_session_reuse off;
        proxy_pass_header Server;
        proxy_cache_bypass $http_upgrade;
        proxy_redirect off;
    }
}
